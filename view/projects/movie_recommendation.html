                        # Import necessary libraries
                        import pandas as pd
                        from scipy.stats import pearsonr

                        # Load the movie ratings dataset
                        ratings = pd.read_csv('ratings.csv')

                        # Create a user-movie matrix with NaN for missing values
                        user_movie_matrix = ratings.pivot_table(index='user_id', columns='movie_id', values='rating')

                        # Function to calculate Pearson correlation between two users
                        def pearson_correlation(user1, user2):
                            common_movies = user_movie_matrix.loc[user1].notna() & user_movie_matrix.loc[user2].notna()
                            
                            if common_movies.sum() == 0:
                                return 0  # No common movies, correlation is undefined
                            
                            # Calculate Pearson correlation coefficient
                            correlation_coefficient, _ = pearsonr(
                                user_movie_matrix.loc[user1, common_movies],
                                user_movie_matrix.loc[user2, common_movies]
                            )
                            
                            return correlation_coefficient

                        # Function to get top N movie recommendations for a user
                        def get_movie_recommendations(user, N=5):
                            # Find users with similar preferences using Pearson correlation
                            similar_users = [(other_user, pearson_correlation(user, other_user)) for other_user in user_movie_matrix.index if user != other_user]
                            
                            # Sort users by correlation coefficient in descending order
                            similar_users.sort(key=lambda x: x[1], reverse=True)
                            
                            # Get top N similar users
                            top_users = similar_users[:N]
                            
                            # Initialize a dictionary to store movie recommendations and their scores
                            recommendations = {}
                            
                            # Iterate through top users and their ratings
                            for other_user, correlation in top_users:
                                movies_not_watched = user_movie_matrix.loc[user].isna() & user_movie_matrix.loc[other_user].notna()
                                
                                # Multiply ratings by correlation coefficients and sum
                                weighted_ratings = user_movie_matrix.loc[other_user, movies_not_watched] * correlation
                                recommendations.update(weighted_ratings)
                            
                            # Sort recommendations by score in descending order
                            sorted_recommendations = sorted(recommendations.items(), key=lambda x: x[1], reverse=True)
                            
                            # Get the top N movie recommendations
                            top_recommendations = [movie for movie, _ in sorted_recommendations[:N]]
                            
                            return top_recommendations
